Q-1=> Create 2 objects - parent and child. Attach methods to parent and use those methods in child object using parents prototype. 
ans1:-
            function Parent(name) {
                this.name = name;
              }
              
              Parent.prototype.sayHello = function() {
                console.log(`Hello, my name is ${this.name}.`);
              };
              
              function Child(name, age) {
                Parent.call(this, name);
                this.age = age;
              }
              
              Child.prototype = Object.create(Parent.prototype);
              
              Child.prototype.introduce = function() {
                console.log(`I am ${this.name} and I am ${this.age} years old.`);
              };
              
              const parentObj = new Parent("Parent");
              const childObj = new Child("Child", 5);
              
              parentObj.sayHello(); 
              childObj.sayHello(); 
              childObj.introduce();

Q-2=>Write code to explain prototype chaining 
ans2:-
            function Parent(name) {
                this.name = name;
              }
              
              Parent.prototype.greet = function() {
                console.log(`Hello, my name is ${this.name}`);
              };
              
              function Child(name, age) {
                Parent.call(this, name);
                this.age = age;
              }
              
              Child.prototype = Object.create(Parent.prototype);
              Child.prototype.constructor = Child;
              
              Child.prototype.showAge = function() {
                console.log(`${this.name} is ${this.age} years old.`);
              };
              
              const parentObj = new Parent('John');
              const childObj = new Child('Alice', 5);
              
              parentObj.greet(); 
              childObj.greet();  
              childObj.showAge();

Q-3 Add a method to calculate sum of all elements in Array in array's protype, use that method to calculate sum for multiple arrays 
ans3:-
            function calculateArraySum() {
                let sum = 0;
                for (let i = 0; i < this.length; i++) {
                  sum += this[i];
                }
                return sum;
              }
              Array.prototype.calculateSum = calculateArraySum;
              const arr1 = [1, 2, 3, 4, 5];
              const arr2 = [10, 20, 30];
              const arr3 = [2, 4, 6, 8, 10];
              
              console.log(arr1.calculateSum()); // Output: 15 (1 + 2 + 3 + 4 + 5)
              console.log(arr2.calculateSum()); // Output: 60 (10 + 20 + 30)
              console.log(arr3.calculateSum()); // Output: 30 (2 + 4 + 6 + 8 + 10)

Q-4 Write a JavaScript function to retrieve all the names of object's own and inherited properties. 
ans4:-
            function getAllPropertyNames(obj) {
                const propertyNames = [];
              
                function getAllPropertyNamesRecursive(obj) {
                  if (obj === null || obj === undefined) return;
              
                  const ownPropertyNames = Object.getOwnPropertyNames(obj);
                  propertyNames.push(...ownPropertyNames);
              
                  const prototype = Object.getPrototypeOf(obj);
              
                  getAllPropertyNamesRecursive(prototype);
                }
              
                getAllPropertyNamesRecursive(obj);
              
                return Array.from(new Set(propertyNames));
              }
